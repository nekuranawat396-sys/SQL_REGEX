-- between operator find the range of value
select *from actor where actor_id between 4 and 6; -- cannot bw use before higher value first upon lower value

-- in operator find the specifice value 
select *from actor where actor_id in (4 ,7);

-- like operators 
select*from actor where first_name like 'A%';
select*from actor where first_name like '%ed';

select*from actor where first_name like '%A%';

-- At sny place
